Оператор case
Ключевое слово Case обеспечивает структурированный эквивалент последовательности условных операторов той же самой переменной.
Оператор case более изящен, более эффективен, и его проще обслуживать чем множество вложений if. 

Синтаксис оператора case
case Ordinal expression of
    Ordinal value {,Ordinal value...} : Statement;
        {...}
    else
        Statement;
end;

Пример
program Menu;

var
  C: Integer;
  I: Integer;
  X, F: Real;

begin
  WriteLn('Select action:');
  WriteLn('1: Calculate Sin(X)');
  WriteLn('2: Calculate Cos(X)');
  WriteLn('3: Calculate Sqr(X)');
  WriteLn('4: Print stars line');
  Write('Your choise: ');
  ReadLn(C);
  case C of
    1,2,3:
    begin
      Write('Enter X: ');
      ReadLn(X);
      case C of
        1: F := Sin(X);
        2: F := Cos(X);
        3: F := Sqr(X);
      end;
      WriteLn('X = ', X:0:4, ', F = ', F:0:4);
    end;
    4:
    begin
      for I := 1 to 80 do
        Write('*');
      WriteLn;
    end;
    else
      WriteLn('Incorrect action');
  end;
  ReadLn;
end.
